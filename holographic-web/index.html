Behaviors 


var Spin = function (config) {
    var object3d;
    function awake(o) {
        object3d = o;
    }
    function update(deltaTime) {
        object3d.rotation.y += config.speed * deltaTime;
    }
    return { awake: awake, update: update };
};

cube.addBehavior(
    Spin({speed:0.002}),
);
