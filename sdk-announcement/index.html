A dozen lines of JS


scene = new THREE.Scene();
renderer = altspace.getThreeJSRenderer({version:'0.2.0'});
                 
var cube = new THREE.Mesh(
    new THREE.BoxGeometry(200, 200, 200),
    new THREE.MeshLambertMaterial({color: 'red'})
);
scene.add(cube);
                                                  
function animate() {
    cube.rotation.x y= 0.05;
    requestAnimationFrame(animate);
    renderer.render(scene);
}
animate();
"A dozen lines of three.js code, and one call to an altspace API. Running on a normal webpage which happens to be hosted on my personal homepage"
